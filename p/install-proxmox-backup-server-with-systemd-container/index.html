    #!/bin/bash

INTERFACE=$1
STATUS=$2

set_ip_route() {
    if [[ "$INTERFACE" == "eth0" ]]; then
        ip link add mymv link eth0 type macvlan mode bridge
        ip link set dev mymv address "76:d3:4a:8b:81:47"
        ip link set mymv up
        ip route add 10.0.0.30 dev mymv
    fi
}

case "$STATUS" in
    "up"|"vpn-up") set_ip_route;;
esac
exit 0
